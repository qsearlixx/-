    function updateOptions() {

        const family = document.getElementById('familyType').value;

        const roomSelect = document.getElementById('roomType');

        roomSelect.innerHTML = '';

        for (let room in db[family]) {

            let opt = document.createElement('option');

            opt.value = room;

            opt.innerHTML = room.replace('_', ' ');

            roomSelect.appendChild(opt);

        }

        calculate();

    }


    function calculate() {

        const dateInput = document.getElementById('arrivalDate').value;

        const dateVal = new Date(dateInput);

        const familySelect = document.getElementById('familyType');

        const family = familySelect.value;

        const room = document.getElementById('roomType').value;

        const weekSelect = document.getElementById('weeks');

        const weekIdx = weekSelect.value;

        

        const month = dateVal.getMonth();

        const day = dateVal.getDate();

        let isHighSeason = (month > 4 && month < 9) || (month === 4) || (month === 9 && day <= 15);


        const priceTypeIdx = isHighSeason ? 1 : 0;

        const finalPrice = db[family][room][weekIdx][priceTypeIdx];

        document.getElementById('total').innerText = finalPrice.toLocaleString();


        // Данные для бланка

        document.getElementById('summary').innerHTML = `

            <p><strong>Клиент:</strong> ${document.getElementById('clientName').value || '________________'}</p>

            <p><strong>Период заезда:</strong> ${dateInput} (${isHighSeason ? 'Летний сезон' : 'Стандартный сезон'})</p>

            <p><strong>Программа:</strong> ${familySelect.options[familySelect.selectedIndex].text}</p>

            <p><strong>Категория номера:</strong> ${room}</p>

            <p><strong>Длительность:</strong> ${weekSelect.options[weekSelect.selectedIndex].text}</p>

        `;

        

        document.getElementById('seasonIndicator').innerText = isHighSeason ? "☀ Летние цены (с 01.05 по 15.10)" : "❄ Стандартные цены";

        document.getElementById('seasonIndicator').className = "season-badge " + (isHighSeason ? "summer" : "winter");

        document.getElementById('promoNote').style.display = family.includes('free') ? 'block' : 'none';

    }


    function syncPrint() { calculate(); }
